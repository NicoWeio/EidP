#include <iostream>
using namespace std;

void verdopple(int &num) {
	num *= 2;
}

int main() {
	unsigned int const n = 8;
	int zahl[n] = { 5, 44, -13, 78, -14, 66, 78, 9 };
	cout << "Zahlen vor der Verdopplung " << endl;
	for (unsigned int i = 0; i < n; ++i) {
		cout << (i + 1) << ". Zahl: " << zahl[i] << endl;
	}

	for (unsigned int i = 0; i < n; ++i) {
		verdopple(zahl[i]);
	}

	cout << "Zahlen nach Verdopplung " << endl;
	for (unsigned int i = 0; i < n; ++i) {
		cout << (i + 1) << ". Zahl: " << zahl[i] << endl;
	}
	return 0;
}
